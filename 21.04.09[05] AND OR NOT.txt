SELECT *
FROM EMPLOYEE
WHERE DNO = 10 AND JOB = 'MANAGER';


SELECT *
FROM EMPLOYEE
WHERE DNO = 10 OR JOB = 'MANAGER';


SELECT *
FROM EMPLOYEE
WHERE NOT DNO = 10


SELECT *
FROM EMPLOYEE
WHERE HIREDATE < '1982/01/01' AND JOB = 'CLERK';


SELECT *
FROM EMPLOYEE
WHERE SALARY >= 1500 AND JOB = 'SALESMAN';


SELECT *
FROM EMPLOYEE
WHERE DNO = 20 OR DNO = 30
ORDER BY ENAME ASC;
