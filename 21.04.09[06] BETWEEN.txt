SELECT *
FROM EMPLOYEE
WHERE NOT SALARY BETWEEN 1000 AND 1500;


SELECT *
FROM EMPLOYEE
WHERE NOT SALARY NOT BETWEEN 1000 AND 1500;


SELECT *
FROM EMPLOYEE
WHERE SALARY < 1000 OR SALARY > 1500


SELECT ENO, ENAME, HIREDATE
FROM EMPLOYEE
WHERE HIREDATE BETWEEN '1981/04/01' AND '1981/09/30';


SELECT ENAME, HIREDATE
FROM EMPLOYEE
WHERE HIREDATE BETWEEN '80/12/31' AND '82/01/01';


SELECT ENAME, SALARY
FROM EMPLOYEE
WHERE NOT SALARY BETWEEN 2000 AND 3000

FROM EMPLOYEE
