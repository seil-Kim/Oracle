SELECT *
FROM EMPLOYEE
WHERE COMMISSION IN(300, 500, 1400);


SELECT *
FROM EMPLOYEE
WHERE DNO IN(20,30)
ORDER BY ENAME ASC;


SELECT ENAME, DNO
FROM EMPLOYEE
WHERE DNO IN(20,30)
ORDER BY ENAME;


SELECT ENAME, JOB, SALARY
FROM EMPLOYEE
WHERE JOB IN('CLERK', 'SALESMAN') AND SALARY NOT IN(1600, 950, 1300);
