SELECT *
FROM EMPLOYEE
ORDER BY SALARY ASC;


SELECT *
FROM EMPLOYEE
ORDER BY SALARY;


SELECT *
FROM EMPLOYEE
ORDER BY SALARY DESC, ENAME ASC;


SELECT *
FROM EMPLOYEE
WHERE COMMISSION > 0
ORDER BY COMMISSION DESC;


SELECT *
FROM EMPLOYEE
WHERE JOB = 'MANAGER'
ORDER BY HIREDATE ASC;


SELECT ENO, ENAME, (SALARY * 12 + NVL(COMMISSION, 0)) AS 연봉
FROM EMPLOYEE
WHERE (SALARY * 12 + NVL(COMMISSION, 0)) >= 17000
ORDER BY ENO ASC;


SELECT ENO, ENAME, (SALARY * 12 + NVL(COMMISSION, 0)) AS 연봉
FROM EMPLOYEE
WHERE (SALARY * 12 + NVL(COMMISSION, 0)) >= 17000
ORDER BY 연봉 ASC;


SELECT ENAME, SALARY, (SALARY * 12 + 100) AS "연간 총수입"
FROM EMPLOYEE
ORDER BY "연간 총수입" DESC;


SELECT ENAME, SALARY
FROM EMPLOYEE
WHERE SALARY > 2000
ORDER BY SALARY DESC;


SELECT ENAME, SALARY, COMMISSION
FROM EMPLOYEE
WHERE COMMISSION IS NOT NULL
ORDER BY SALARY DESC, COMMISSION DESC;
